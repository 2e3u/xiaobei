控制结构

掌握C语言中的条件语句和循环结构

if-else 条件语句

根据条件执行不同的代码块。

基本语法

// 简单if语句
if (score >= 60) {
    printf("及格\n");
}

// if-else语句
if (age >= 18) {
    printf("成年人\n");
} else {
    printf("未成年\n");
}

// if-else if-else语句
if (score >= 90) {
    printf("优秀\n");
} else if (score >= 80) {
    printf("良好\n");
} else if (score >= 60) {
    printf("及格\n");
} else {
    printf("不及格\n");
}

switch-case 语句

示例代码

int day = 3;
switch (day) {
    case 1:
        printf("星期一\n");
        break;
    case 2:
        printf("星期二\n");
        break;
    case 3:
        printf("星期三\n");
        break;
    default:
        printf("其他日期\n");
}

注意事项

  • switch只能用于整数类型的判断
  • 每个case后需要加break防止穿透
  • default是可选的默认分支

for循环

基本用法

// 基本for循环
for (int i = 0; i < 5; i++) {
    printf("%d ", i);
}

// 循环遍历数组
int arr[] = {1, 2, 3, 4, 5};
for (int i = 0; i < sizeof(arr)/sizeof(arr[0]); i++) {
    printf("%d ", arr[i]);
}

// 嵌套for循环（打印乘法表）
for (int i = 1; i <= 9; i++) {
    for (int j = 1; j <= i; j++) {
        printf("%d×%d=%d ", j, i, i*j);
    }
    printf("\n");
}

while和do-while循环

示例代码

// while循环
int count = 0;
while (count < 5) {
    printf("%d ", count);
    count++;
}

// do-while循环（至少执行一次）
int num;
do {
    printf("请输入一个正数: ");
    scanf("%d", &num);
} while (num <= 0);

控制语句

break和continue

// break示例
for (int i = 0; i < 10; i++) {
    if (i == 5) break;  // 到5就结束循环
    printf("%d ", i);
}

// continue示例
for (int i = 0; i < 10; i++) {
    if (i % 2 == 0) continue;  // 跳过偶数
    printf("%d ", i);
}
